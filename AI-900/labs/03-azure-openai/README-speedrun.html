Lab 3 — Speedrun

az login
az group create -n rg-ai900-oai -l eastus
az cognitiveservices account create -n ai900-oai-demo -g rg-ai900-oai --kind OpenAI --sku S0 -l eastus --yes
az cognitiveservices account deployment create -n ai900-oai-demo -g rg-ai900-oai \
  --deployment-name chat --model-name gpt-4o-mini --model-version "2024-07-18" \
  --model-format OpenAI --sku-name Standard --sku-capacity 1

ENDPOINT=$(az cognitiveservices account show -n ai900-oai-demo -g rg-ai900-oai --query properties.endpoint -o tsv)
KEY=$(az cognitiveservices account keys list -n ai900-oai-demo -g rg-ai900-oai --query key1 -o tsv)

cd labs/03-azure-openai
python -m venv .venv && source .venv/bin/activate
pip install -r requirements.txt

AZURE_OAI_ENDPOINT=$ENDPOINT AZURE_OAI_KEY=$KEY AZURE_OAI_DEPLOYMENT=chat \
  python chat.py "Explain RAG in two sentences."

az group delete -n rg-ai900-oai --yes --no-wait
