Lab 1 — Speedrun

Goal: analyze an image with Azure AI Vision. Prereq: Azure CLI + Python venv ready.

az login
az group create -n rg-ai900-vision -l eastus
az cognitiveservices account create -n ai900-vision-demo -g rg-ai900-vision --kind ComputerVision --sku F0 -l eastus --yes
ENDPOINT=$(az cognitiveservices account show -n ai900-vision-demo -g rg-ai900-vision --query properties.endpoint -o tsv)
KEY=$(az cognitiveservices account keys list -n ai900-vision-demo -g rg-ai900-vision --query key1 -o tsv)

cd labs/01-azure-ai-vision
python -m venv .venv && source .venv/bin/activate
pip install -r requirements.txt
AZURE_AI_VISION_ENDPOINT=$ENDPOINT AZURE_AI_VISION_KEY=$KEY python analyze.py https://learn.microsoft.com/azure/ai-services/computer-vision/media/quickstarts/presentation.png

# clean up
az group delete -n rg-ai900-vision --yes --no-wait
